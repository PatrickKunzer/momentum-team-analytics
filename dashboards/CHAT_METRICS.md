## SQL Queries

### Messages per Hour

```sql
SELECT
  DATE_TRUNC('hour', timestamp) as hour,
  COUNT(*) as messages,
  COUNT(DISTINCT user_id) as unique_users
FROM events
WHERE event = 'message_sent'
  AND timestamp > NOW() - INTERVAL 24 HOUR
GROUP BY hour
ORDER BY hour
```

### Agent Usage Distribution

```sql
SELECT
  properties.agent_id,
  properties.agent_name,
  COUNT(*) as messages,
  COUNT(DISTINCT user_id) as unique_users,
  ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER (), 1) as percentage
FROM events
WHERE event = 'message_sent'
  AND timestamp > NOW() - INTERVAL 30 DAY
GROUP BY properties.agent_id, properties.agent_name
ORDER BY messages DESC
```

```sql
  PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY properties.response_time_ms) as p50,
  PERCENTILE_CONT(0.95) WITHIN GROUP (ORDER BY properties.response_time_ms) as p95,
  PERCENTILE_CONT(0.99) WITHIN GROUP (ORDER BY properties.response_time_ms) as p99
FROM events
WHERE event = 'response_received'
  AND timestamp > NOW() - INTERVAL 24 HOUR
```

### Feature Usage in Messages

```sql
SELECT
  'with_attachments' as feature,
  COUNT(CASE WHEN properties.has_attachments = true THEN 1 END) * 100.0 / COUNT(*) as percentage
FROM events WHERE event = 'message_sent' AND timestamp > NOW() - INTERVAL 30 DAY

UNION ALL

SELECT
  'with_knowledge' as feature,

UNION ALL

```

### Feedback Analysis

```sql
SELECT
  properties.agent_id,
  COUNT(CASE WHEN properties.feedback_type = 'thumbs_up' THEN 1 END) as thumbs_up,
  COUNT(CASE WHEN properties.feedback_type = 'thumbs_down' THEN 1 END) as thumbs_down,
  ROUND(
    COUNT(CASE WHEN properties.feedback_type = 'thumbs_up' THEN 1 END) * 100.0 /
    NULLIF(COUNT(*), 0),
    1
  ) as positive_rate
FROM events
WHERE event = 'message_feedback_given'
  AND timestamp > NOW() - INTERVAL 30 DAY
GROUP BY properties.agent_id
ORDER BY (thumbs_up + thumbs_down) DESC
```
